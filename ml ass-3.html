import matplotlib.pyplot as plt
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

# Read the data from the CSV file
data = pd.read_csv('Eye_Glaucoma.csv') 
df = pd.DataFrame(data)

# Convert 'DOB' column to datetime
df['DOB'] = pd.to_datetime(df['DOB'])
df['Age_At_Surgery'] = df['Age_At_Surgery']

# Extracting the year from DOB
df['DOB_year'] = df['DOB'].dt.year 

# Scatter plot with blue data points
plt.scatter(df['DOB_year'], df['Age_At_Surgery'], color='blue', label='Data Points')
plt.xlabel('DOB Year')
plt.ylabel('Age at Surgery')
plt.title('Linear Regression of DOB vs. Age at Surgery')

# Linear regression
regression = LinearRegression().fit(np.array(df['DOB_year']).reshape(-1, 1), df['Age_At_Surgery'])

# Plotting the regression line with red color
plt.plot(df['DOB_year'], regression.predict(np.array(df['DOB_year']).reshape(-1, 1)), color='red', label='Regression Line')

# Display legend
plt.legend()

# Show the plot
plt.show()
